Section 2 Prepare Climate Data for Interpolation

In this script, we start by curating and processing the climate data. We interpolate the dataset to a fine resolution for future analysis.

2.1 Load libraries

library(tidyverse) 
library(lubridate)
library(terra)
library(sf)

2.2 Prepare climate data

Read data and add date. Please contact the lead author for a copy of this data. We will add this to Zenodo as a release upon publication.

clim_files <- list.files("data/climate/historical", full.names = T)
data <- map(clim_files, read_delim,
  delim = " ", col_names = c("ppt", "tmin", "tmax", "wind")
)

# seq dates
seq_date <- seq(
  ymd("1870-01-01"),
  ymd("2018-12-31"),
  by = "days"
)

# add date, year, month, and season
data <- map2(
  data, clim_files,
  function(df, clf) {

    # get the coordinates
    coords <- str_extract_all(clf, "\\d+\\.\\d+")[[1]]
    mutate(
      df,
      # add the date, year, and month, and season
      # dry season is december to may 12 - 5
      date = seq_date,
      year = year(date),
      month = month(date),
      season = if_else(between(month, 6, 11), "rainy", "dry"),

      # convert coordinates
      x = as.numeric(coords[2]),
      y = as.numeric(coords[1])
    )
  }
)

# bind all data
data <- bind_rows(data)

# check data
assertthat::assert_that(
  nrow(
    distinct(data, x, y)
  ) == length(clim_files),
  msg = "error in some weather station coordinates"
)

# filter for southern western ghats
# y < 12 and y > 9
data <- filter(
  data,
  between(y, 9, 12)
)

# save data
write_csv(
  data,
  file = "results/climate/data_clim_daily.csv"
)

2.3 Summarise by month

# get daily mean temperature
data <- mutate(
  data,
  tmean = (tmax + tmin) / 2
)

# prepare to summarise
data <- group_by(
  data,
  x, y, month, year, season,
) |>
  # groups will be added by default
  select(
    ppt, tmean
  )

# summarise monthly ppt sum and tmean mean and sd
data <- summarise(
  data,
  ppt = sum(ppt),
  t_mean = mean(tmean),
  t_sd = sd(tmean)
)

# save summary
write_csv(
  data,
  "results/climate/data_monthly_climate.csv"
)

2.4 Prepare for spatial data

# get unique data, i.e., from unique coords
coords <- ungroup(data) |>
  distinct(x, y) |>
  mutate_all(
    as.numeric
  )

2.5 Prepare spatial data

# get unique coordinates
coord_sf <- st_as_sf(
  coords,
  coords = c("x", "y"),
  crs = 4326
)

# get extent to clip elevation
ext <- st_bbox(coord_sf) |>
  st_as_sfc() |>
  st_transform(32643) |>
  st_buffer(25000) |>
  st_transform(4326)

# load elevation raster
elevation <- terra::rast("data/elevation/alt")
# crop by extent of study area
elevation_hills <- terra::crop(elevation, as(ext, "Spatial"))

# save a smaller piece
terra::writeRaster(
  elevation_hills,
  "results/elevation/raster_elevation.tif",
  overwrite = TRUE
)

2.6 Prepare coastline data

# load coastline and elevation
coast <- st_read("data/landcover/India_Boundary.shp")
coast_line <- st_cast(coast, "MULTILINESTRING")

# load small elevation raster
elev <- terra::rast("results/elevation/raster_elevation.tif")
elev_utm <- terra::project(elev, "epsg:32643")

2.7 Resample raster to 600m

# resample the raster to 600 m from 30m
elev_utm_600 <- rast(ext(elev_utm), resolution = 600)
elev_utm_600 <- terra::resample(
  elev_utm, elev_utm_600
)
crs(elev_utm_600) <- "epsg:32643"

# recast to 4326
elev_600 <- terra::project(elev_utm_600, "epsg:4326")
# make coast raster
# the below lines of code take a very long time to run
coast_raster <- terra::distance(elev_600, terra::vect(coast_line))
coast_raster <- terra::mask(coast_raster, terra::vect(coast))
# latitude raster using copy of elevation and assigning coordinate
lat_raster <- data.table::copy(elev_600)
terra::values(lat_raster) <- terra::yFromCell(
  lat_raster,
  cell = seq(length(values(lat_raster)))
)
# stack rasters
stack <- c(coast_raster, elev_600, lat_raster)
names(stack) <- c("coast", "elev", "lat")
terra::writeRaster(
  stack,
  "results/climate/raster_gam_stack.tif",
  overwrite = T
)

2.8 Link predictors to climate data

# extract covariates using high-res raster
coords <- mutate(
  coords,

  # extract elevation
  elev = terra::extract(
    elevation_hills,
    st_coordinates(coord_sf)
  )$alt,

  # get distance to coastline
  coast = as.numeric(
    st_distance(coord_sf, coast_line)
  ),

  # get latitude
  lat = y
)

# link to climate data summary
data <- left_join(
  data,
  coords
)

# bin into 10 year intervals
data <- mutate(
  data,
  year_bin = floor(year / 10) * 10
)

# save data
write_csv(
  data,
  "results/climate/data_for_gam.csv"
)
